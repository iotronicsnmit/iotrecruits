// Firestore Security Rules for IoT Recruit (Free Plan Compatible)
// Copy these rules to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow read/write access to students collection
    match /students/{studentPhone} {
      allow read, write: if true; // Simplified for free plan - you can add auth later
    }
    
    // Allow read/write access to game attempts
    match /gameAttempts/{attemptId} {
      allow read, write: if true; // Simplified for free plan
    }
    
    // Allow read/write access to leaderboards
    match /leaderboards/{gameType} {
      allow read, write: if true; // Simplified for free plan
      
      // Allow access to scores subcollection
      match /scores/{scoreId} {
        allow read, write: if true; // Simplified for free plan
      }
    }
    
    // Allow read/write access to test collection (for connection testing)
    match /test/{testDoc} {
      allow read, write: if true; // Simplified for free plan
    }
    
    // Deny all other collections by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
